CREATE TABLE current_lecture (
    previous_lect_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    lect_rename VARCHAR(10),
    cmp_div ENUM('전필', '전선', '교필', '교선', '일선'),
    is_cyber TINYINT,
    credit INT,
    subject_type ENUM('교양', '전공'),
    semester_year VARCHAR(10),
    lect_id VARCHAR(15),
    notice TEXT,
    professor VARCHAR(20)
);


CREATE TABLE previous_lecture (
    previous_lect_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    lect_name VARCHAR(255),
    cmp_div VARCHAR(255),
    is_cyber TINYINT,
    credit INT,
    subject_type VARCHAR(255),
    semester_year VARCHAR(10),
    code VARCHAR(255),
    notice TEXT,
    professor VARCHAR(255)
);
